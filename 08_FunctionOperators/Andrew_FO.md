capture_all <- function(f){
  force(f)
  function(...) {
    out <- list()
    out$ouput <- capture.output(f(...))
    out$error <- tryCatch(f(...),error = function(e) e)
    out$warnin <- tryCatch(f(...),warning = function(w) w)
    out
    }
  }

maker <- function(f){
  force(f)

    lapply(list(...), force)

    function(x) {

    }
